经典的二分查找，做这道题的时间都用在了二分查找的几个细节上。


Position BinarySearch(List L, ElementType X) {
	int begin= 0, end = L->Last;
	
	while (begin<=end)
	{
		int mid = (begin + end) / 2;

		if (L->Data[mid] == X)
		{
			return mid;
		}
		else if (L->Data[mid]>X)
		{
			end = mid - 1;
		}
		else // mid < X
		{
			begin = mid + 1;
		}
	}

	return NotFound;
}
