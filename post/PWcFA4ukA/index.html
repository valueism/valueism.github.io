using  node  = struct Node;

	tree right;

void read(vector<int>& pre, vector<int>& in) {
	stack<int> stk;
	int n, m;
	cin >> n;
	m = n * 2;
	while (m--)
	{
		string str;//"Push" or "Pop"
		cin >> str;
		if (str=="Push")
		{
			int a;
			cin >> a;
			pre.push_back(a);
			stk.push(a);
		}
		if (str=="Pop")
		{
			int p = stk.top();
			stk.pop();
			in.push_back(p);
		}

	}

}
